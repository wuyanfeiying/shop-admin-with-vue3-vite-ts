```
shop-admin-with-vue3-vite-ts
├─ .gitignore // git忽略文件

├─ package-lock.json
├─ package.json // 依赖包
├─ public // 静态资源

├─ src
│  ├─ api // 后端服务API

│  ├─ router // 路由
│  ├─ store // 状态管理
│  ├─ style.css // 全局样式
│  ├─ styles // 样式
│  ├─ utils // 工具
│  ├─ views // 视图

├─ tsconfig.node.json
```
